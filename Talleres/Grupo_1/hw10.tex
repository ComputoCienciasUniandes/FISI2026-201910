\begin{lstlisting}[language=Python]
import numpy as np
import matplotlib.pyplot as plt
from scipy.optimize import curve_fit

### Cargar y visualizar datos

data = np.loadtxt("dos_picos_1.dat")
x = data[:,0]
y = data[:,1]
plt.scatter(x,y)


### Ajuste de curvas

def f(x, a, b):
    y = a + b * x
    return y

curve_fit(f,x,y)
\end{lstlisting}
